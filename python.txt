import requests
from bs4 import BeautifulSoup
import csv
import json
import time
import logging
import re
import sys

logging.basicConfig(
    filename='scraper.log',
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)

def clean_price(price_text):
        try:
            price = re.sub(r'[^\d.]', '', price_text)  
            return float(price)
        except:
            return None
    
def scrape_page(url):
        try:
            response = requests.get(url, timeout=5)
            response.raise_for_status()
        except requests.exceptions.RequestException as e:
            logging.error(f"Request failed: {e}")
            return []
    
        soup = BeautifulSoup(response.text, 'html.parser')
        books = soup.find_all('article', class_='product_pod')
    
        data = []
    
        for book in books:
            try:
                name = book.h3.a['title']
                price = clean_price(book.find('p', class_='price_color').text)
                rating = book.p['class'][1]
    
                data.append({
                    "name": name,
                    "price": price,
                    "rating": rating
                })
            except Exception as e:
                logging.warning(f"Error parsing book: {e}")
        
        return data

def save_csv(data, filename="books.csv"):
        try:
            with open(filename, 'w', newline='', encoding='utf-8') as file:
                writer = csv.DictWriter(file, fieldnames=["name", "price", "rating"])
                writer.writeheader()
                writer.writerows(data)
            print(f"[✔] Data saved to {filename}")
        except Exception as e:
            logging.error(f"CSV save error: {e}")
    
def save_json(data, filename="books.json"):
        try:
            with open(filename, 'w', encoding='utf-8') as file:
                json.dump(data, file, indent=4)
            print(f"[✔] Data saved to {filename}")
        except Exception as e:
            logging.error(f"JSON save error: {e}")
    
def main():
        print("=== Python Web Scraper ===")
    
        try:
            start_page = int(input("Enter start page: "))
            end_page = int(input("Enter end page: "))
        except:
            print("Invalid input. Using default pages 1–3.")
            start_page, end_page = 1, 3
    
        base_url = "http://books.toscrape.com/catalogue/page-{}.html"
        all_data = []
    
        for page in range(start_page, end_page + 1):
            url = base_url.format(page)
            print(f"Scraping page {page}...")
    
            page_data = scrape_page(url)
            all_data.extend(page_data)
    
            time.sleep(1)  # polite delay
    
        if not all_data:
            print("No data scraped. Exiting.")
            sys.exit()
    
        save_csv(all_data)
        save_json(all_data)
    
        print(f"\nTotal records scraped: {len(all_data)}")

if __name__ == "__main__":
    main()
